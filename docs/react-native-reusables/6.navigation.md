### 6. A. 3. Custom List Item

```jsx
import * as React from 'react';

        </Text>

});
```

```jsx
  NavigationMenu,
  NavigationMenuContent,
  NavigationMenuItem,
  NavigationMenuLink,
  NavigationMenuList,
  NavigationMenuTrigger,
  navigationMenuTriggerStyle,

import { Text } from '~/components/ui/text';

  {
    title: 'Alert Dialog',
    href: '/alert-dialog',

  },
  {
    title: 'Button',
    href: '/button',
    description: 'Button component with various styles and states.',
  },
  {
    title: 'Card',
    href: '/card',
    description: 'Container for displaying content and actions related to a single subject.',
  },

                  <Text>
                    Re-usable components built using Radix UI on the web and Tailwind CSS.
                  </Text>
                </ListItem>

                <ListItem href='/docs/installation' title='Installation'>
                  <Text>How to install dependencies and structure your app.</Text>
                </ListItem>

                <ListItem href='/docs/typography' title='Typography'>

      </NavigationMenu>
    </>
  );
}
```

```jsx
import { useRouter } from 'expo-router';
import { Button } from '~/components/ui/button';
import { Text } from '~/components/ui/text';

function NavigationWithExpoRouter() {
```

## 6. C. Tab Navigation

### 6. C. 1. Custom Tab Bar Component

```jsx
import * as React from 'react';
import { View } from 'react-native';
import { useSafeAreaInsets } from 'react-native-safe-area-context';
import { Button } from '~/components/ui/button';
import { Text } from '~/components/ui/text';
import { Home } from '~/lib/icons/Home';
import { User } from '~/lib/icons/User';
import { Settings } from '~/lib/icons/Settings';
import { cn } from '~/lib/utils';

function TabBar({ state, descriptors, navigation }) {
  const insets = useSafeAreaInsets();
  
  return (
    <View
      style={{ paddingBottom: insets.bottom }}
      className="flex-row bg-background border-t border-border"
    >
      {state.routes.map((route, index) => {
        const { options } = descriptors[route.key];
        const label = options.tabBarLabel || options.title || route.name;
        const isFocused = state.index === index;
        
        let IconComponent;
        if (route.name === 'Home') IconComponent = Home;
        else if (route.name === 'Profile') IconComponent = User;
        else if (route.name === 'Settings') IconComponent = Settings;
        
        const onPress = () => {
          const event = navigation.emit({
            type: 'tabPress',
            target: route.key,
            canPreventDefault: true,
          });
          
          if (!isFocused && !event.defaultPrevented) {
            navigation.navigate(route.name);
          }
        };
        
        return (
          <Button
            key={route.key}
            variant="ghost"
            className={cn(
              "flex-1 flex-col h-16 rounded-none",
              isFocused && "bg-secondary"
            )}
            onPress={onPress}
          >
            {IconComponent && (
              <IconComponent 
                size={24}
                className={cn(
                  "text-muted-foreground",
                  isFocused && "text-foreground"
                )}
              />
            )}
            <Text
              className={cn(
                "text-xs text-muted-foreground",
                isFocused && "text-foreground"
              )}
            >
              {label}
            </Text>
          </Button>
        );
      })}
    </View>
  );
}
```

## 6. D. Navigation Drawer

### 6. D. 1. Drawer Component

```jsx
import React, { useState } from 'react';
import { View, Pressable, StyleSheet } from 'react-native';
import { useSafeAreaInsets } from 'react-native-safe-area-context';
import Animated, { 
  useSharedValue, 
  useAnimatedStyle, 
  withTiming,
  Easing 
} from 'react-native-reanimated';
import { Text } from '~/components/ui/text';
import { Button } from '~/components/ui/button';
import { Menu } from '~/lib/icons/Menu';
import { X } from '~/lib/icons/X';

function DrawerExample() {
  const [isOpen, setIsOpen] = useState(false);
  const insets = useSafeAreaInsets();
  const translateX = useSharedValue(-300);
  
  const drawerStyle = useAnimatedStyle(() => {
    return {
      transform: [{ translateX: translateX.value }],
    };
  });
  
  const overlayStyle = useAnimatedStyle(() => {
    return {
      opacity: translateX.value === 0 ? 0.5 : 0,
      display: translateX.value === 0 ? 'flex' : 'none',
    };
  });
  
  const toggleDrawer = () => {
    if (isOpen) {
      translateX.value = withTiming(-300, {
        duration: 300,
        easing: Easing.bezier(0.25, 0.1, 0.25, 1),
      });
    } else {
      translateX.value = withTiming(0, {
        duration: 300,
        easing: Easing.bezier(0.25, 0.1, 0.25, 1),
      });
    }
    setIsOpen(!isOpen);
  };
  
  return (
    <View style={{ flex: 1 }}>
      <Button 
        variant="ghost" 
        onPress={toggleDrawer}
        className="absolute top-4 left-4 z-10"
      >
        <Menu size={24} />
      </Button>
      
      <Animated.View 
        style={[
          StyleSheet.absoluteFill, 
          { backgroundColor: 'black' },
          overlayStyle
        ]}
      >
        <Pressable 
          style={StyleSheet.absoluteFill}
          onPress={toggleDrawer}
        />
      </Animated.View>
      
      <Animated.View 
        style={[
          {
            position: 'absolute',
            top: 0,
            left: 0,
            width: 300,
            height: '100%',
            paddingTop: insets.top,
            paddingBottom: insets.bottom,
            backgroundColor: 'var(--background)',
            zIndex: 20,
            borderRightWidth: 1,
            borderRightColor: 'var(--border)',
          },
          drawerStyle
        ]}
      >
        <View className="flex-row justify-between items-center p-4 border-b border-border">
          <Text className="text-xl font-bold">Menu</Text>
          <Button variant="ghost" onPress={toggleDrawer}>
            <X size={24} />
          </Button>
        </View>
        
        <View className="p-4">
          <Button 
            variant="ghost"
            className="justify-start w-full mb-2"
          >
            <Text>Home</Text>
          </Button>
          <Button 
            variant="ghost"
            className="justify-start w-full mb-2"
          >
            <Text>Profile</Text>
          </Button>
          <Button 
            variant="ghost"
            className="justify-start w-full mb-2"
          >
            <Text>Settings</Text>
          </Button>
          <Button 
            variant="ghost"
            className="justify-start w-full mb-2"
          >
            <Text>Help</Text>
          </Button>
        </View>
      </Animated.View>
    </View>
  );
}
```
